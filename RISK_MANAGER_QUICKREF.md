### Option 2: Manual Configuration
```python
from agents.risk_manager import RiskManager, RiskConfig

# Create custom config
config = RiskConfig(

```

## Position Sizing

### Basic Usage
```python
from agents.risk_manager import SignalInput
from decimal import Decimal

# Create signal
signal = SignalInput(
    signal_id="sig_001",
    symbol="BTC/USD",
    side="long",
    entry_price=Decimal("50000.00"),
    stop_loss=Decimal("49000.00"),  # 2% SL
    take_profit=Decimal("52000.00"),
    confidence=Decimal("0.75"),
)

# Size position
equity = Decimal("10000.00")
position = rm.size_position(signal, equity)

# Check result
if position.allowed:
    print(f"Position allowed:")
    print(f"  Size: {position.size} BTC")
    print(f"  Notional: ${position.notional_usd}")
    print(f"  Risk: ${position.expected_risk_usd} ({float(position.risk_pct):.2%})")
    print(f"  Leverage: {position.leverage}x")
else:
    print(f"Position rejected: {position.rejection_reasons}")
```

## Portfolio Risk Check

### Check Current Risk
```python
# Get all current/proposed positions
positions = [position1, position2, position3]

# Check portfolio risk
result = rm.check_portfolio_risk(positions, equity)

if result.passed:
    print(f"Portfolio risk OK:")
    print(f"  Total risk: {float(result.total_risk_pct):.2%}")
    print(f"  Positions: {result.position_count}")
else:
    print(f"Portfolio violations:")
    for violation in result.violations:
        print(f"  - {violation}")
```

### Before Adding Position
```python
# Get current positions
current_positions = get_active_positions()

# Size new position
new_position = rm.size_position(signal, equity)

# Check if adding new position would exceed limits
all_positions = current_positions + [new_position]
risk_check = rm.check_portfolio_risk(all_positions, equity)

if risk_check.passed and new_position.allowed:
    # Safe to add position
    execute_position(new_position)
else:
    # Reject due to portfolio risk
    print(f"Cannot add position: {risk_check.violations}")
```

## Drawdown Breaker

### Update Drawdown State
```python
from decimal import Decimal

# Equity curve (list of equity values)
equity_curve = [
    Decimal("10000"),  # Bar 0
    Decimal("10100"),  # Bar 1
    Decimal("9750"),   # Bar 2 (down -3.5% from peak)
]

# Update state
current_bar = 2
dd_state = rm.update_drawdown_state(equity_curve, current_bar)

print(f"Drawdown state:")
print(f"  Daily DD: {dd_state.daily_dd_pct:.2%}")
print(f"  Rolling DD: {dd_state.rolling_dd_pct:.2%}")
print(f"  Mode: {dd_state.mode}")
print(f"  Risk multiplier: {dd_state.risk_multiplier}")
print(f"  Pause remaining: {dd_state.pause_remaining} bars")
if dd_state.trigger_reason:
    print(f"  Reason: {dd_state.trigger_reason}")
```

### Get Current State
```python
# Get current drawdown state without updating
current_state = rm.get_drawdown_state()

if current_state.mode == "hard_halt":
    print("Trading halted due to drawdown")
elif current_state.mode == "soft_stop":
    print(f"Risk reduced to {current_state.risk_multiplier}x")
else:
    print("Normal trading conditions")
```

### Drawdown Modes

| Mode | Trigger | Risk Multiplier | Behavior |
|------|---------|----------------|----------|
| `normal` | No DD breach | 1.0 | Full risk (2% per trade) |
| `soft_stop` | Daily DD ≤ -2% | 0.5 | Reduced risk (1% per trade) |
| `hard_halt` | Rolling DD ≤ -5% | 0.0 | No new positions for N bars |

## Common Patterns

### Pattern 1: Complete Risk Flow
```python
from agents.risk_manager import create_risk_manager, SignalInput
from decimal import Decimal

# Setup
rm = create_risk_manager(
    per_trade_risk_pct=0.02,
    max_portfolio_risk_pct=0.04,
    leverage_caps={"BTC/USD": 3.0}
)

equity = Decimal("10000.00")
equity_curve = [equity]  # Track for DD monitoring

# Signal from strategy router
signal = SignalInput(...)

# 1. Update drawdown state
dd_state = rm.update_drawdown_state(equity_curve, current_bar=0)

if dd_state.mode == "hard_halt":
    print("Trading halted")
    # Skip position sizing
else:
    # 2. Size position
    position = rm.size_position(signal, equity)

    if position.allowed:
        # 3. Check portfolio risk
        all_positions = get_current_positions() + [position]
        risk_check = rm.check_portfolio_risk(all_positions, equity)

        if risk_check.passed:
            # 4. Execute position
            execute_position(position)
        else:
            print(f"Portfolio risk exceeded: {risk_check.violations}")
    else:
        print(f"Position rejected: {position.rejection_reasons}")
```

### Pattern 3: Drawdown Recovery
```python
# Track equity over time
equity_curve = []

for bar in trading_bars:
    current_equity = get_current_equity()
    equity_curve.append(current_equity)

    # Update DD state
    dd_state = rm.update_drawdown_state(equity_curve, current_bar=bar)

    if dd_state.mode == "hard_halt":
        if dd_state.pause_remaining > 0:
            print(f"Halted: {dd_state.pause_remaining} bars remaining")
            continue  # Skip trading
        else:
            print("Cooldown expired, checking DD...")
            if dd_state.rolling_dd_pct > -0.05:
                print("Recovered, resuming trading")
            else:
                print("Still in drawdown, extending halt")

    elif dd_state.mode == "soft_stop":
        print(f"Risk reduced to {dd_state.risk_multiplier}x")
        # Positions will be sized with 0.5x risk
```

### Pattern 4: Dynamic Leverage Adjustment
```python
# Load leverage caps from exchange config
import yaml

with open("config/exchange_configs/kraken.yaml") as f:
    kraken_config = yaml.safe_load(f)

# Extract leverage caps
leverage_caps = {}
margin_config = kraken_config.get("trading_specs", {}).get("margin", {})
max_leverage_map = margin_config.get("max_leverage", {})
symbol_map = kraken_config.get("symbols", {}).get("denormalize", {})

for kraken_symbol, leverage in max_leverage_map.items():
    if kraken_symbol != "default":
        internal_symbol = symbol_map.get(kraken_symbol, kraken_symbol)
        leverage_caps[internal_symbol] = leverage

# Create risk manager with loaded caps
rm = create_risk_manager(leverage_caps=leverage_caps)

# Positions will respect per-symbol caps
btc_position = rm.size_position(btc_signal, equity)  # Max 5x (BTC cap)
eth_position = rm.size_position(eth_signal, equity)  # Max 5x (ETH cap)
```

## Configuration Reference

### Full RiskConfig Fields
```python
class RiskConfig:
    # Per-trade risk
    per_trade_risk_pct_min: float = 0.01       # Min 1% risk
    per_trade_risk_pct_max: float = 0.02       # Max 2% risk

    # Portfolio caps
    max_portfolio_risk_pct: float = 0.04       # Max 4% total risk
    max_concurrent_positions: int = 3          # Max 3 positions

    # Leverage
    default_leverage: float = 2.0              # Default 2x
    max_leverage_default: float = 5.0          # Max 5x
    leverage_caps: Dict[str, float] = {}       # Per-symbol caps

    # Drawdown thresholds
    dd_daily_threshold_pct: float = -0.02      # Daily DD threshold (-2%)
    dd_rolling_threshold_pct: float = -0.05    # Rolling DD threshold (-5%)
    dd_rolling_window_bars: int = 20           # Rolling window (20 bars)

    # Drawdown actions
    dd_risk_multiplier_soft: float = 0.5       # Soft stop (0.5x risk)
    dd_pause_bars: int = 10                    # Hard halt pause (10 bars)

    # Position size filter
    min_position_usd: float = 10.0             # Min position size ($10)
```

## Troubleshooting

### Position Rejected (below_min_position_usd)
**Problem**: Position size too small
```python
position = rm.size_position(signal, equity)
# position.rejection_reasons = ["below_min_position_usd"]
```

**Causes**:
- Very wide SL (large SL distance)
- Very small equity
- Large drawdown multiplier applied

**Solutions**:
1. Tighten stop loss
2. Increase equity
3. Adjust `min_position_usd` config
4. Wait for DD recovery

### Portfolio Risk Exceeded
**Problem**: Total risk > 4%
```python
risk_check = rm.check_portfolio_risk(positions, equity)
# risk_check.violations = ["portfolio_risk_exceeded: 6.00% > 4.00%"]
```

**Solutions**:
1. Close/reduce existing positions
2. Reject new signal
3. Increase equity
4. Adjust `max_portfolio_risk_pct` config

### Max Positions Exceeded
**Problem**: Too many concurrent positions
```python
risk_check = rm.check_portfolio_risk(positions, equity)
# risk_check.violations = ["max_positions_exceeded: 4 > 3"]
```

**Solutions**:
1. Close weakest position first
2. Reject new signal
3. Adjust `max_concurrent_positions` config

### Drawdown Hard Halt Not Expiring
**Problem**: Cooldown not decrementing
```python
# WRONG: Not calling update_drawdown_state every bar
if should_trade:
    dd_state = rm.update_drawdown_state(equity_curve, bar)

# CORRECT: Call every bar
dd_state = rm.update_drawdown_state(equity_curve, bar)  # Decrements cooldown
```

**Solution**: Always call `update_drawdown_state()` on every bar, even when not trading.

### Leverage Not Applied Correctly
**Problem**: Leverage differs from expected

**Check**:
1. Symbol leverage cap: `rm.get_max_leverage(symbol)`
2. Config default: `rm.config.default_leverage`
3. Config max: `rm.config.max_leverage_default`

**Logic**:
```python
symbol_cap = rm.get_max_leverage(symbol)     # e.g., 3.0 for BTC
max_leverage = min(symbol_cap, config.max_leverage_default)  # min(3.0, 5.0) = 3.0
leverage = min(config.default_leverage, max_leverage)        # min(2.0, 3.0) = 2.0
```

## Performance

- **Position sizing**: O(1) per signal (~1ms)
- **Portfolio check**: O(N) where N = position count (~1ms for N=10)
- **Drawdown update**: O(W) where W = window size (~2ms for W=20)
- **Memory**: O(1) per RiskManager instance
